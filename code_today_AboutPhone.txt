class myCustomContainer extends StatelessWidget {
  IconData? icon;
  String? name;
  String? value;

  myCustomContainer({required this.icon, this.name, this.value});

  Widget build(BuildContext context) {
    return Container(
      //color: Colors.black,
      child: Column(
        children: [Icon(icon), Text('$name'), Text('$value')],
      ),
    );
  }
}
